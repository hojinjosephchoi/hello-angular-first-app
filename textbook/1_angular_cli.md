# install cli
~~~
npm install -g @angular/cli
~~~

# angular cli 버전확인
~~~
ng --version
~~~

# 애플리케이션 생성
~~~
ng new {app-name}
~~~

# 애플리케이션 서버 실행
~~~
ng serve
~~~

# 라우트/컴포넌트 등의 blueprint를 만듦
~~~
ng generate
~~~

# 명령어 help
~~~
ng --help
~~~

# 프로젝트 주요 소스 구성
- src/ : 션 소스구성
- src/app : 애플리케이션 관련 폴더
- src/assets : 이미지 등
- src/environments : 개발모드/프로덕션모드 등 환경변수
- src/main.ts : 메인파일
